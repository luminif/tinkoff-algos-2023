using namespace std;

void solve() {
    string s;
    vector<string> a;

    while (cin >> s) {
        a.push_back(s);
    }

    sort(all(a), [&](string x, string y) -> bool {
         return x + y > y + x;
    });

    for (auto &t : a) cout << t;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int t = 1;
    //cin >> t;

    while (t--) {
        solve();
    }

    //cerr << "runtime: " << clock() / 1000.0 << '\n';
    return 0;
}
